\section*{QCE 3.5}

\begin{align*}
\begin{pmatrix}
0 & 1 \\
1 & 0
\end{pmatrix}
\end{align*}

\begin{align*}
\begin{pmatrix}
0 & 1 \\
1 & 0
\end{pmatrix}
\begin{pmatrix}
a \\
b
\end{pmatrix} =
\lambda \begin{pmatrix}
a \\
b
\end{pmatrix}.
\end{align*}
Now
\begin{align*}
\begin{pmatrix}
b \\
a
\end{pmatrix} = \lambda \begin{pmatrix}
a \\
b
\end{pmatrix},
\end{align*}
so $b = \lambda a$ and $a = \lambda b$. If $\lambda = 1$, the corresponding eigenvector is $(1, 1)^T$. If $\lambda = 0$, the eigenvector is $(0, 0)^T$.

\section*{QCE 3.7}
\begin{align*}
\begin{pmatrix}
1 & 0 & 2 \\
0 & 3 & 4 \\
1 & 0 & 2
\end{pmatrix} \begin{pmatrix}
a \\
b \\
c \\
\end{pmatrix} = \begin{pmatrix}
a + 2c \\
3b + 4c \\
a + 2c
\end{pmatrix} = 
\begin{pmatrix}
\lambda a \\
\lambda b \\
\lambda c
\end{pmatrix}.

\section*{QCE 3.8}
Suppose
\[
A = \begin{pmatrix}
a_{1, 1} & \dots & a_{1,n} \\
\vdots & \ddots & \vdots \\
a_{n, 1} & \dots & a_{n, n}
\end{pmatrix},
B = \begin{pmatrix}
b_{1, 1} & \dots & b_{1,n} \\
\vdots & \ddots & \vdots \\
b_{n, 1} & \dots & b_{n, n}
\end{pmatrix}.
\]
 Now, it is easy to see that 
\begin{align*}
Tr(A + B) &= (a_{1, 1} + b_{1, 1}) + \dots + (a_{n,n} + b_{n, n}) \\
               &= (a_{1,1} + \dots + a_{n,n})  + (b_{1,1} + \dots + b_{n,n}) \\
               &= Tr(A) + Tr(B).
\end{align*}
Also
\begin{align*}
  Tr(\lambda A) &= \lambda a_{1,1} + \dots + \lambda a_{n,n} \\
                        &= \lambda Tr(A).
\end{align*}
Also-also
\begin{align*}
Tr(AB) &= a_{1, 1}b_{1,1} + \dots + a_{n,n}b_{n,n} \\
          &= b_{1, 1}a_{1,1} + \dots + b_{n,n}a_{n,n} \\
          &= Tr(BA),
\end{align*}
as real multiplication commutes.
